export const setMentors = (state = [], action) => {
  switch(action.type) {
 
    case 'SET_MENTORS':
      return action.mentors;
 
    case 'UPDATE_CHANGED_MENTOR':
 
      return state.map(mentor => {
        if (mentor.id === action.mentor.id) {
          return action.mentor;
        } else {
          return mentor
        }
      })
 
    default:
      return state;
  }
}
